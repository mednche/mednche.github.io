Data visualisation CFS - Detroit¶

This notebook provides a data exploration of the Detroit call-for-service (CFS) dataset.

In particular, it explores:

- the police's response times (intake, dispatch and travel)
- the types of calls 
- the priority of calls
- the temporal trends of calls

# Get temporal dataframe
def get_time_df(data_heatmap):
    # convert column to timestamp
    data_heatmap['Call Time'] = pd.to_datetime(data_heatmap['Call Time'])

    dff = pd.DataFrame({'month': data_heatmap['Call Time'].dt.month, 'dow': data_heatmap['Call Time'].dt.day_name(), 
                        'hour': data_heatmap['Call Time'].dt.hour})
    
    return dff

# convert Hour to int so it can be ordered for the graph


dff_all = get_time_df(data)
dff_p1 = get_time_df(data_p1)
